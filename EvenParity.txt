

CHIP EvenParity {
 IN bit0, bit1, bit2, bit3, bit4, bit5, bit6, bit7;
    OUT out;

    PARTS:
       Xor(a=bit0,b=bit1,out=A0);
       Xor(a=bit2,b=bit3,out=A1);
       Xor(a=bit4,b=bit5,out=A2);
       Xor(a=bit6,b=bit7,out=A3);
       
       Xor(a=A0,b=A1,out=B0);
       Xor(a=A2,b=A3,out=B1);
       Xor(a=B0,b=B1,out=out);
}
